1. 회원 통계 (회원 테이블: user)
총 회원 수
SELECT COUNT(DISTINCT user_id) AS total_members
FROM user;

신규 가입자 수 (예: 2025년 5월 1일 이후)
SELECT COUNT(DISTINCT user_id) AS new_members
FROM user
WHERE created_at >= '2025-05-01';

신고된 회원 수 (reported 컬럼이 'Y'인 경우)
SELECT COUNT(DISTINCT user_id) AS reported_members
FROM user
WHERE reported = 'Y';

reported 컬럼이 없을 경우, 신고 테이블(report_user) 따로 만들어서 JOIN 처리
-- 예: report_user 테이블이 따로 있는 경우
SELECT COUNT(DISTINCT u.user_id) AS reported_members
FROM user u
JOIN report_user r ON u.user_id = r.target_user_id;

2. 펀딩 통계 (펀딩 테이블: funding)
누적 펀딩 금액
SELECT SUM(funding_money) AS total_funding
FROM funding;

이번 달 펀딩 금액 (2025년 5월 기준)
SELECT SUM(funding_money) AS this_month_funding
FROM funding
WHERE MONTH(created_at) = 5 AND YEAR(created_at) = 2025;

3. 프로젝트 통계 (프로젝트 테이블: project)
전체 프로젝트 수
SELECT COUNT(*) AS total_projects
FROM project;
진행 중 프로젝트 수 (status = 'ACTIVE' 등)
SELECT COUNT(*) AS active_projects
FROM project
WHERE status = 'ACTIVE';

정산 대기 프로젝트 수 (settlement_status = 'WAITING')
SELECT COUNT(*) AS pending_settlement_projects
FROM project
WHERE settlement_status = 'WAITING';
