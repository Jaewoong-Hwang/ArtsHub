1. 회원 목록 조회 (검색 + 필터 적용)
SELECT user_id, name, email, role, status, created_at
FROM user
WHERE 1 = 1
  AND (name LIKE CONCAT('%', #{keyword}, '%') OR email LIKE CONCAT('%', #{keyword}, '%'))
  AND (status = #{status} OR #{status} IS NULL)
  AND (role = #{role} OR #{role} IS NULL)
ORDER BY created_at DESC;

2. 회원 상태별 통계
SELECT 
  SUM(CASE WHEN status = 'ACTIVE' THEN 1 ELSE 0 END) AS activeCount,
  SUM(CASE WHEN status = 'PENDING' THEN 1 ELSE 0 END) AS pendingCount,
  SUM(CASE WHEN status = 'SUSPENDED' THEN 1 ELSE 0 END) AS suspendedCount,
  SUM(CASE WHEN status = 'WITHDRAWN' THEN 1 ELSE 0 END) AS withdrawnCount,
  SUM(CASE WHEN status = 'DORMANT' THEN 1 ELSE 0 END) AS dormantCount
FROM user;
